第12章：JavaWeb 实战项目（用户管理系统）

✅ 项目说明：
项目名称：用户管理系统（User Manage System）
技术栈：SpringBoot3.2 + MyBatis + MySQL8 + SpringMVC + 文件上传
核心功能：登录认证、用户CRUD、头像上传、登录拦截、全局统一返回
全程适配：IDEA2025社区版 + JDK24 + 内嵌Tomcat11，无会员、纯手动

12.3.2 统一返回结果（Result 工具类）

package com.ums.util;

import lombok.Data;

@Data
public class Result {
    private int code;
    private String msg;
    private T data;

    // 成功
    public static  Result success(T data) {
        Result result = new Result<>();
        result.setCode(200);
        result.setMsg("操作成功");
        result.setData(data);
        return result;
    }

    // 失败
    public static  Result fail(String msg) {
        Result result = new Result<>();
        result.setCode(500);
        result.setMsg(msg);
        result.setData(null);
        return result;
    }
}

12.3.7 Service 实现类（UserServiceImpl）

package com.ums.service.impl;

import com.ums.mapper.UserMapper;
import com.ums.pojo.User;
import com.ums.service.UserService;
import com.ums.util.Result;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.web.multipart.MultipartFile;
import java.io.File;
import java.util.UUID;

@Service
public class UserServiceImpl implements UserService {

    @Autowired
    private UserMapper userMapper;

    // 登录
    @Override
    public Result login(String username, String password) {
        User user = userMapper.login(username, password);
        if(user == null) return Result.fail("用户名或密码错误");
        return Result.success(user);
    }

    // 查询所有
    @Override
    public Result findAll() {
        return Result.success(userMapper.findAll());
    }

    // 添加
    @Override
    public Result add(User user) {
        return userMapper.add(user) > 0 ? Result.success(null) : Result.fail("添加失败");
    }

    // 修改
    @Override
    public Result update(User user) {
        return userMapper.update(user) > 0 ? Result.success(null) : Result.fail("修改失败");
    }

    // 删除
    @Override
    public Result delete(Integer id) {
        return userMapper.delete(id) > 0 ? Result.success(null) : Result.fail("删除失败");
    }

    // 头像上传
    @Override
    public Result uploadAvatar(MultipartFile file) {
        try {
            // 生成唯一文件名
            String fileName = UUID.randomUUID() + "_" + file.getOriginalFilename();
            // 上传路径
            String path = System.getProperty("user.dir") + "/src/main/webapp/upload/";
            File dir = new File(path);
            if(!dir.exists()) dir.mkdirs();
            // 写入文件
            file.transferTo(new File(dir, fileName));
            // 返回文件路径
            return Result.success("/upload/" + fileName);
        } catch (Exception e) {
            e.printStackTrace();
            return Result.fail("上传失败");
        }
    }
}

12.3.8 登录拦截器（LoginInterceptor）

package com.ums.interceptor;

import com.ums.util.Result;
import com.fasterxml.jackson.databind.ObjectMapper;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import org.springframework.web.servlet.HandlerInterceptor;

public class LoginInterceptor implements HandlerInterceptor {
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
        Object user = request.getSession().getAttribute("loginUser");
        if(user == null) {
            response.setContentType("application/json;charset=utf-8");
            response.getWriter().write(new ObjectMapper().writeValueAsString(Result.fail("请先登录")));
            return false;
        }
        return true;
    }
}

12.3.9 拦截器配置（MvcConfig）

package com.ums.config;

import com.ums.interceptor.LoginInterceptor;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
import org.springframework.web.servlet.config.annotation.ResourceHandlerRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

@Configuration
public class MvcConfig implements WebMvcConfigurer {
    // 登录拦截
    @Override
    public void addInterceptors(InterceptorRegistry registry) {
        registry.addInterceptor(new LoginInterceptor())
                .addPathPatterns("/**")
                .excludePathPatterns("/user/login", "/upload/**", "/error");
    }

    // 静态资源映射
    @Override
    public void addResourceHandlers(ResourceHandlerRegistry registry) {
        registry.addResourceHandler("/upload/**")
                .addResourceLocations("file:" + System.getProperty("user.dir") + "/src/main/webapp/upload/");
    }
}

12.3.10 Controller 层（UserController）

package com.ums.controller;

import com.ums.pojo.User;
import com.ums.service.UserService;
import com.ums.util.Result;
import jakarta.servlet.http.HttpSession;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.*;
import org.springframework.web.multipart.MultipartFile;

@RestController
@RequestMapping("/user")
public class UserController {

    @Autowired
    private UserService userService;

    // 登录
    @PostMapping("/login")
    public Result login(String username, String password, HttpSession session) {
        Result result = userService.login(username, password);
        if(result.getCode() == 200) session.setAttribute("loginUser", result.getData());
        return result;
    }

    // 查询所有用户
    @GetMapping("/list")
    public Result list() {
        return userService.findAll();
    }

    // 添加用户
    @PostMapping("/add")
    public Result add(@RequestBody User user) {
        return userService.add(user);
    }

    // 修改用户
    @PutMapping("/update")
    public Result update(@RequestBody User user) {
        return userService.update(user);
    }

    // 删除用户
    @DeleteMapping("/delete/{id}")
    public Result delete(@PathVariable Integer id) {
        return userService.delete(id);
    }

    // 头像上传
    @PostMapping("/upload")
    public Result upload(MultipartFile file) {
        return userService.uploadAvatar(file);
    }
}

12.4 项目启动 & 接口测试

启动步骤
  1. 创建 src/main/webapp/upload 文件夹（存储头像）
  2. 运行启动类 SpringbootUmsApplication
  3. 控制台出现启动成功提示即完成

接口清单（Postman 直接测试）

  • 登录：POST http://localhost:8080/user/login?username=admin&password=123456
  • 用户列表：GET http://localhost:8080/user/list
  • 添加用户：POST http://localhost:8080/user/add（JSON参数）
  • 头像上传：POST http://localhost:8080/user/upload（form-data文件）
  • 修改/删除：PUT/DELETE 请求
❗ 测试必看：
1. 必须先登录，否则所有接口被拦截
2. 上传文件夹必须手动创建
3. MySQL 服务必须启动

12.5 项目部署（SpringBoot 一键打包）

打包 & 运行步骤（IDEA2025社区版）
  1. 右侧打开 Maven → Lifecycle → package
  2. 等待打包完成，在 target 目录生成 xxx.jar
  3. 打开 cmd，进入 jar 所在目录
  4. 运行命令：java -jar springboot-ums-1.0-SNAPSHOT.jar
  5. 项目启动，内嵌Tomcat11直接运行
✅ 优势：不用外置Tomcat，不用配置服务器，直接运行JAR包即可上线
