SSM 深层复习与进阶（Spring+SpringMVC+MyBatis）

一、SSM 核心知识点深层复盘

SSM是企业级JavaWeb开发的基础框架组合，先巩固核心本质，再攻克进阶难点：

技术栈 核心本质 核心注解/配置 企业级痛点（进阶解决）
Spring IOC容器 + DI依赖注入 + AOP面向切面 @Component/@Service/@Autowired、@Aspect、@Transactional 事务管理、Bean生命周期管控、多环境适配
SpringMVC 前端控制器 + 请求映射 + 视图解析 @Controller/@RequestMapping/@ResponseBody 全局异常处理、参数校验、拦截器链、文件上传优化
MyBatis SQL映射 + 结果集自动封装 @Mapper、#{}/${}、动态SQL、ResultMap 分页、缓存、多数据源、SQL性能优化
SSM整合 三层架构 + 配置整合 多XML配置、Spring整合MyBatis 配置繁琐、事务一致性、扩展性差
✅ 核心认知：SSM的本质是解耦 - 控制层/业务层/数据层分离，Spring作为容器统一管理所有组件。

二、Spring 进阶（企业级核心）

2.1 Bean 进阶管控（解决「怎么造、怎么管」）

2.1.1 Bean 作用域（实战必懂）

作用域 核心说明 企业使用场景 代码示例
singleton（默认） 整个应用只有一个实例 无状态组件（Service/Mapper/工具类） @Service（默认单例）
prototype 每次获取都是新实例 有状态组件（购物车、线程不安全对象） @Service + @Scope("prototype")
request 一次HTTP请求一个实例 请求级数据封装 @Scope("request")
session 一次会话一个实例 用户登录信息存储 @Scope("session") 
// 多例Bean实战示例（购物车）

@Scope("prototype") // 关键：设置为多例
public class CartService {

    private Integer userId;
    private List productIds = new ArrayList<>();
    
    // 添加商品
    public void addProduct(Integer productId) {
        productIds.add(productId);
    }
}

2.2.2 事务使用（业务层注解）

// 2. 业务层使用事务
package com.ssm.service.impl;

import com.ssm.mapper.UserMapper;
import com.ssm.pojo.User;
import com.ssm.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Isolation;
import org.springframework.transaction.annotation.Propagation;
import org.springframework.transaction.annotation.Transactional;

import java.util.List;

@Service
public class UserServiceImpl implements UserService {

    @Autowired
    private UserMapper userMapper;

    /**
     * 批量新增用户 - 事务控制
     * propagation：传播行为（REQUIRED=有事务就加入，无则新建）
     * isolation：隔离级别（READ_COMMITTED=避免脏读，企业标配）
     * rollbackFor：指定异常回滚（默认只回滚RuntimeException）
     */
    @Transactional(
        propagation = Propagation.REQUIRED,
        isolation = Isolation.READ_COMMITTED,
        rollbackFor = Exception.class // 所有异常都回滚
    )
    @Override
    public void batchAddUser(List userList) {
        for (User user : userList) {
            userMapper.addUser(user);
            // 模拟异常：第二个用户插入失败，全部回滚
            if (user.getId() == 2) {
                throw new RuntimeException("插入失败，事务回滚");
            }
        }
    }
}
✅ 事务核心参数：
1. propagation：REQUIRED（默认）/REQUIRES_NEW（新建事务）/NESTED（嵌套事务）
2. isolation：READ_COMMITTED（企业标配）/REPEATABLE_READ（MySQL默认）
3. rollbackFor：必须指定Exception.class，否则检查型异常不回滚

2.3 Spring AOP 进阶（切面编程）

核心场景：日志记录、权限校验、性能监控（无侵入式增强）。

// 1. AOP切面类
package com.ssm.aspect;

import org.aspectj.lang.ProceedingJoinPoint;
import org.aspectj.lang.annotation.Around;
import org.aspectj.lang.annotation.Aspect;
import org.springframework.stereotype.Component;

@Aspect // 标记为切面
@Component // 交给Spring管理
public class LogAspect {

    /**
     * 环绕通知：性能监控
     * 切点表达式：execution(* com.ssm.service.*.*(..)) 匹配service包下所有方法
     */
    @Around("execution(* com.ssm.service.*.*(..))")
    public Object monitorPerformance(ProceedingJoinPoint joinPoint) throws Throwable {
        // 前置：记录开始时间
        long startTime = System.currentTimeMillis();
        
        // 执行原方法
        Object result = joinPoint.proceed();
        
        // 后置：计算耗时
        long costTime = System.currentTimeMillis() - startTime;
        String methodName = joinPoint.getSignature().getName();
        System.out.println("方法[" + methodName + "]执行耗时：" + costTime + "ms");
        
        return result;
    }
}

// 2. 全局异常处理器
package com.ssm.exception;

import com.ssm.util.Result;
import org.springframework.validation.BindException;
import org.springframework.web.bind.MethodArgumentNotValidException;
import org.springframework.web.bind.annotation.ExceptionHandler;
import org.springframework.web.bind.annotation.RestControllerAdvice;

// @RestControllerAdvice = @ControllerAdvice + @ResponseBody
@RestControllerAdvice
public class GlobalExceptionHandler {

    // 处理业务异常
    @ExceptionHandler(BusinessException.class)
    public Result handleBusinessException(BusinessException e) {
        return Result.fail(e.getMessage());
    }

    // 处理参数校验异常
    @ExceptionHandler({MethodArgumentNotValidException.class, BindException.class})
    public Result handleValidException(MethodArgumentNotValidException e) {
        String msg = e.getBindingResult().getFieldError().getDefaultMessage();
        return Result.fail("参数校验失败：" + msg);
    }

    // 处理所有运行时异常
    @ExceptionHandler(RuntimeException.class)
    public Result handleRuntimeException(RuntimeException e) {
        e.printStackTrace(); // 日志记录
        return Result.fail("服务器异常：" + e.getMessage());
    }

    // 处理所有异常（兜底）
    @ExceptionHandler(Exception.class)
    public Result handleException(Exception e) {
        e.printStackTrace();
        return Result.fail("系统异常，请联系管理员");
    }
}

// 3. 统一返回结果工具类
package com.ssm.util;

import lombok.Data;

@Data
public class Result {
    private int code; // 200成功，500失败
    private String msg;
    private T data;

    // 成功响应
    public static  Result success(T data) {
        Result result = new Result<>();
        result.setCode(200);
        result.setMsg("操作成功");
        result.setData(data);
        return result;
    }

    // 失败响应
    public static  Result fail(String msg) {
        Result result = new Result<>();
        result.setCode(500);
        result.setMsg(msg);
        result.setData(null);
        return result;
    }
}

3.3 拦截器链（多拦截器执行顺序）

// 1. 第一个拦截器：登录校验
package com.ssm.interceptor;

import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import org.springframework.web.servlet.HandlerInterceptor;

public class LoginInterceptor implements HandlerInterceptor {
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
        // 登录校验逻辑
        Object user = request.getSession().getAttribute("loginUser");
        if (user == null) {
            response.getWriter().write("请先登录");
            return false; // 拦截
        }
        return true; // 放行
    }
}

// 2. 第二个拦截器：权限校验
package com.ssm.interceptor;

import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import org.springframework.web.servlet.HandlerInterceptor;

public class PermissionInterceptor implements HandlerInterceptor {
    @Override
    public boolean preHandle(HttpServletRequest request, HttpServletResponse response, Object handler) throws Exception {
        // 权限校验逻辑
        String role = (String) request.getSession().getAttribute("role");
        if (!"admin".equals(role)) {
            response.getWriter().write("无操作权限");
            return false;
        }
        return true;
    }
}

// 3. 配置拦截器链（指定执行顺序）
package com.ssm.config;

import com.ssm.interceptor.LoginInterceptor;
import com.ssm.interceptor.PermissionInterceptor;
import org.springframework.context.annotation.Configuration;
import org.springframework.web.servlet.config.annotation.InterceptorRegistry;
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer;

@Configuration
public class WebConfig implements WebMvcConfigurer {

    @Override
    public void addInterceptors(InterceptorRegistry registry) {
        // 拦截器执行顺序：先添加的先执行
        registry.addInterceptor(new LoginInterceptor())
                .addPathPatterns("/**")
                .excludePathPatterns("/user/login");
        
        registry.addInterceptor(new PermissionInterceptor())
                .addPathPatterns("/user/**")
                .excludePathPatterns("/user/login", "/user/query");
    }
}

package com.ssm.config;

import com.github.pagehelper.PageHelper;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

import java.util.Properties;

@Configuration
public class MyBatisConfig {
    @Bean
    public PageHelper pageHelper() {
        PageHelper pageHelper = new PageHelper();
        Properties props = new Properties();
        props.setProperty("helperDialect", "mysql"); // 数据库方言
        props.setProperty("reasonable", "true"); // 页码越界自动处理
        props.setProperty("supportMethodsArguments", "true");
        pageHelper.setProperties(props);
        return pageHelper;
    }
}

package com.ssm.service.impl;

import com.github.pagehelper.PageHelper;
import com.github.pagehelper.PageInfo;
import com.ssm.mapper.UserMapper;
import com.ssm.pojo.User;
import com.ssm.service.UserService;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

4.3 自定义TypeHandler（类型转换）

解决：数据库字段类型与Java类型不匹配（如数据库VARCHAR存JSON，Java转对象）。

// 1. 自定义TypeHandler
package com.ssm.typehandler;

import com.alibaba.fastjson2.JSONObject;
import org.apache.ibatis.type.BaseTypeHandler;
import org.apache.ibatis.type.JdbcType;

import java.sql.CallableStatement;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;

// 处理JSON字符串 ↔ JSONObject
public class JsonTypeHandler extends BaseTypeHandler {

    // 设置参数（Java → 数据库）
    @Override
    public void setNonNullParameter(PreparedStatement ps, int i, JSONObject parameter, JdbcType jdbcType) throws SQLException {
        ps.setString(i, parameter.toJSONString());
    }

    // 获取结果（数据库 → Java）
    @Override
    public JSONObject getNullableResult(ResultSet rs, String columnName) throws SQLException {
        return JSONObject.parseObject(rs.getString(columnName));
    }

    @Override
    public JSONObject getNullableResult(ResultSet rs, int columnIndex) throws SQLException {
        return JSONObject.parseObject(rs.getString(columnIndex));
    }

    @Override
    public JSONObject getNullableResult(CallableStatement cs, int columnIndex) throws SQLException {
        return JSONObject.parseObject(cs.getString(columnIndex));
    }
}

六、SSM 进阶核心总结

✅ Spring 进阶核心：
1. Bean管控：作用域（prototype解决线程安全）、生命周期（初始化/销毁资源）；
2. 事务管理：@Transactional（传播行为+隔离级别+回滚规则）；
3. AOP：切面编程（日志/监控/权限，无侵入增强）。
✅ SpringMVC 进阶核心：
1. 全局异常：@RestControllerAdvice统一返回，避免try-catch；
2. 参数校验：@Validated+JSR380注解，替代手动if；
3. 拦截器链：多拦截器顺序配置，实现登录+权限双层校验。
✅ MyBatis 进阶核心：
1. 分页：PageHelper一行代码实现分页，企业必用；
2. 缓存：一级缓存默认开启，二级缓存慎用（频繁修改表禁用）；
3. 类型转换：TypeHandler解决特殊类型映射（JSON/枚举）。
✅ 整合进阶核心：
1. 纯注解替代XML，减少配置文件；
2. 适配JDK24+Jakarta，替换javax为jakarta包；
3. 事务一致性：Spring事务+MyBatis操作，保证数据原子性。
