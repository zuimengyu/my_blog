一、事件处理基础 (从零开始)

1.1 什么是事件？

事件就是用户与网页交互时发生的动作，比如：

  • 点击按钮 (click)
  • 鼠标移动 (mousemove)
  • 键盘按键 (keydown)
  • 表单提交 (submit)
  • 页面加载完成 (load)

1.2 事件绑定的三种方式

方式 写法 优点 缺点
HTML内联 <button onclick="fn()"> 简单直接 结构与行为耦合
DOM属性 btn.onclick = fn 结构行为分离 同一事件只能绑定一个函数
事件监听 btn.addEventListener() 可绑定多个函数，可移除 写法稍复杂 

// 方式1：HTML内联 (不推荐)
// <button onclick="alert('点击了！')">点击我</button>

// 方式2：DOM属性绑定
const btn1 = document.getElementById('btn1');
btn1.onclick = function() {
    alert('通过DOM属性绑定的事件！');
};

// 方式3：事件监听 (推荐！)
const btn2 = document.getElementById('btn2');
btn2.addEventListener('click', function() {
    alert('通过addEventListener绑定的事件！');
});

// 可以绑定多个函数
btn2.addEventListener('click', function() {
    console.log('第二个函数也执行了！');
});

// 移除事件监听
function handleClick() {
    alert('这个事件可以被移除');
}
btn2.addEventListener('click', handleClick);
// btn2.removeEventListener('click', handleClick); // 移除监听

二、事件对象和事件流 (进阶)

2.1 事件对象 (event)

当事件触发时，浏览器会自动创建一个事件对象，包含事件的详细信息：


const box = document.getElementById('eventBox');
box.addEventListener('click', function(event) {
    // event 就是事件对象，也常用 e 表示
    console.log('事件类型:', event.type); // click
    console.log('触发元素:', event.target); // 点击的元素
    console.log('绑定元素:', event.currentTarget); // 绑定事件的元素
    console.log('鼠标X坐标:', event.clientX); // 鼠标相对于视口的X坐标
    console.log('鼠标Y坐标:', event.clientY); // 鼠标相对于视口的Y坐标
    
    // 阻止默认行为
    event.preventDefault(); // 比如阻止链接跳转、表单提交
    
    // 阻止事件冒泡
    event.stopPropagation();
});

2.2 事件流 (捕获-目标-冒泡)

事件流描述了事件在DOM树中传播的过程：

  • 捕获阶段：从window向下传播到目标元素
  • 目标阶段：到达目标元素
  • 冒泡阶段：从目标元素向上传播回window

// 事件冒泡演示
const outer = document.getElementById('outer');
const inner = document.getElementById('inner');

outer.addEventListener('click', function() {
    console.log('外层被点击了！');
});

inner.addEventListener('click', function(event) {
    console.log('内层被点击了！');
    // event.stopPropagation(); // 取消注释这行，阻止冒泡
});

// 事件捕获演示 (第三个参数设为true)
outer.addEventListener('click', function() {
    console.log('捕获阶段：外层');
}, true);

inner.addEventListener('click', function() {
    console.log('捕获阶段：内层');
}, true);

2.3 事件委托 (重要！)

利用事件冒泡，给父元素绑定事件来处理子元素的事件，性能更好：


// 不推荐：给每个子元素单独绑定
// const items = document.querySelectorAll('.item');
// items.forEach(item => {
//     item.addEventListener('click', function() {
//         console.log('点击了:', this.innerText);
//     });
// });

// 推荐：事件委托
const list = document.getElementById('itemList');
list.addEventListener('click', function(event) {
    // 检查点击的是否是目标元素
    if(event.target.classList.contains('item')) {
        console.log('点击了:', event.target.innerText);
        // 可以在这里操作被点击的元素
        event.target.style.color = '#4299e1';
    }
});

三、异步编程基础 (回调函数)

3.1 同步 vs 异步

  • 同步：代码按顺序执行，前一个执行完才执行下一个
  • 异步：代码不等待结果，继续执行后面的代码

// 同步代码
console.log('1');
console.log('2');
console.log('3');
// 输出：1, 2, 3

// 异步代码 (setTimeout)
console.log('1');
setTimeout(function() {
    console.log('2');
}, 1000);
console.log('3');
// 输出：1, 3, 2 (2会在1秒后输出)

3.2 回调函数

回调函数就是作为参数传递给另一个函数的函数，在异步操作完成后执行：


// 简单的回调函数
function fetchData(callback) {
    // 模拟异步请求
    setTimeout(function() {
        const data = { name: '张三', age: 18 };
        callback(data); // 异步完成后调用回调函数
    }, 1000);
}

// 使用回调函数
fetchData(function(result) {
    console.log('获取到数据:', result);
});

// 回调地狱 (多个异步操作嵌套)
fetchUser(function(user) {
    fetchOrders(user.id, function(orders) {
        fetchOrderDetails(orders[0].id, function(details) {
            // 嵌套越来越深，代码难以维护
            console.log(details);
        });
    });
});

四、Promise (异步编程的进化)

4.1 Promise基础

Promise是ES6引入的异步编程解决方案，比回调函数更优雅：


// 创建Promise
const promise = new Promise(function(resolve, reject) {
    // 异步操作
    setTimeout(function() {
        const success = true;
        if(success) {
            resolve('操作成功！'); // 成功时调用
        } else {
            reject('操作失败！'); // 失败时调用
        }
    }, 1000);
});

// 使用Promise
promise
    .then(function(result) {
        console.log(result); // 成功时执行
    })
    .catch(function(error) {
        console.log(error); // 失败时执行
    })
    .finally(function() {
        console.log('无论成功失败都会执行');
    });

4.2 Promise链式调用

Promise可以链式调用，解决回调地狱问题：


// 模拟异步函数
function fetchUser() {
    return new Promise((resolve) => {
        setTimeout(() => resolve({ id: 1, name: '张三' }), 1000);
    });
}

function fetchOrders(userId) {
    return new Promise((resolve) => {
        setTimeout(() => resolve([{ id: 101, product: '商品A' }]), 1000);
    });
}

function fetchOrderDetails(orderId) {
    return new Promise((resolve) => {
        setTimeout(() => resolve({ id: orderId, price: 99 }), 1000);
    });
}

// 链式调用 (清晰！)
fetchUser()
    .then(user => {
        console.log('获取用户:', user);
        return fetchOrders(user.id);
    })
    .then(orders => {
        console.log('获取订单:', orders);
        return fetchOrderDetails(orders[0].id);
    })
    .then(details => {
        console.log('获取订单详情:', details);
    })
    .catch(error => {
        console.error('出错了:', error);
    });

4.3 Promise静态方法


// Promise.all - 等待所有Promise完成
Promise.all([fetchUser(), fetchOrders(1)])
    .then(([user, orders]) => {
        console.log('所有数据都获取到了');
    });

// Promise.race - 只要有一个完成就返回
Promise.race([fetchUser(), fetchOrders(1)])
    .then(result => {
        console.log('第一个完成的:', result);
    });

// Promise.allSettled - 等待所有Promise完成(无论成功失败)
Promise.allSettled([fetchUser(), Promise.reject('失败')])
    .then(results => {
        console.log('所有Promise都结束了');
    });

五、async/await (异步编程的终极方案)

5.1 基本用法

async/await是ES2017引入的语法糖，让异步代码看起来像同步代码：


// async函数返回Promise
async function getData() {
    return '数据';
}
getData().then(data => console.log(data));

// await等待Promise完成
async function fetchData() {
    // await只能在async函数中使用
    const result = await new Promise(resolve => {
        setTimeout(() => resolve('等待完成'), 1000);
    });
    console.log(result);
}

5.2 实际应用


// 用async/await重写之前的例子
async function getOrderDetails() {
    try {
        // 顺序执行
        const user = await fetchUser();
        console.log('获取用户:', user);
        
        const orders = await fetchOrders(user.id);
        console.log('获取订单:', orders);
        
        const details = await fetchOrderDetails(orders[0].id);
        console.log('获取订单详情:', details);
        
        return details;
    } catch (error) {
        console.error('出错了:', error);
    }
}

// 并行执行 (提高效率)
async function getAllData() {
    try {
        // 同时发起请求
        const userPromise = fetchUser();
        const ordersPromise = fetchOrders(1);
        
        // 等待所有完成
        const [user, orders] = await Promise.all([userPromise, ordersPromise]);
        
        console.log('用户:', user);
        console.log('订单:', orders);
    } catch (error) {
        console.error('出错了:', error);
    }
}

6.1 案例1：模拟数据加载器

功能：显示加载状态、进度条、错误处理


// 模拟数据加载
async function loadData() {
    const statusEl = document.getElementById('loadingStatus');
    const progressFill = document.getElementById('progressFill');
    const resultEl = document.getElementById('dataResult');
    
    // 显示加载状态
    statusEl.innerHTML = '<span class="loading"></span>正在加载数据...';
    resultEl.innerHTML = '';
    
    try {
        // 模拟进度
        for(let i = 0; i <= 100; i += 10) {
            await new Promise(resolve => setTimeout(resolve, 200));
            progressFill.style.width = i + '%';
        }
        
        // 模拟获取数据
        const data = await new Promise((resolve, reject) => {
            setTimeout(() => {
                // 模拟90%成功率
                if(Math.random() > 0.1) {
                    resolve({
                        success: true,
                        data: [
                            { id: 1, name: '商品A', price: 99 },
                            { id: 2, name: '商品B', price: 199 },
                            { id: 3, name: '商品C', price: 299 }
                        ]
                    });
                } else {
                    reject(new Error('网络请求失败'));
                }
            }, 500);
        });
        
        // 显示结果
        statusEl.innerHTML = '✅ 加载完成！';
        resultEl.innerHTML = `
            <h4>获取到的数据：</h4>
            <ul>
                ${data.data.map(item => `
                    <li>${item.name} - ¥${item.price}</li>
                `).join('')}
            </ul>
        `;
        
    } catch (error) {
        statusEl.innerHTML = '❌ 加载失败：' + error.message;
        progressFill.style.background = '#e53e3e';
    }
}

6.2 案例2：简易聊天机器人

功能：事件处理 + 异步回复

你好！我是聊天机器人，有什么可以帮助你的？

// 聊天机器人核心代码
const chatContainer = document.getElementById('chatContainer');
const chatInput = document.getElementById('chatInput');

// 机器人回复库
const botResponses = {
    '你好': '你好呀！很高兴见到你！',
    '天气': '今天天气晴朗，适合出去走走！',
    '时间': `现在是 ${new Date().toLocaleString()}`,
    '再见': '再见！期待下次聊天！',
    'default': '我不太理解你说的，可以换个话题吗？'
};

// 发送消息
function sendMessage() {
    const message = chatInput.value.trim();
    if(!message) return;
    
    // 添加用户消息
    addMessage(message, 'user');
    chatInput.value = '';
    
    // 显示输入中...
    const typingId = showTyping();
    
    // 模拟机器人思考和回复
    setTimeout(async () => {
        hideTyping(typingId);
        
        // 查找回复
        let response = botResponses['default'];
        for(const key in botResponses) {
            if(message.includes(key)) {
                response = botResponses[key];
                break;
            }
        }
        
        // 模拟打字效果
        await typeMessage(response);
    }, 1000 + Math.random() * 1000);
}

// 添加消息
function addMessage(text, type) {
    const messageDiv = document.createElement('div');
    messageDiv.className = `chat-message ${type}`;
    messageDiv.innerText = text;
    chatContainer.appendChild(messageDiv);
    chatContainer.scrollTop = chatContainer.scrollHeight;
}

// 显示输入中
function showTyping() {
    const typingDiv = document.createElement('div');
    typingDiv.className = 'chat-message bot';
    typingDiv.id = 'typing-indicator';
    typingDiv.innerHTML = '正在输入<span>...</span>';
    chatContainer.appendChild(typingDiv);
    chatContainer.scrollTop = chatContainer.scrollHeight;
    return 'typing-indicator';
}

// 隐藏输入中
function hideTyping(id) {
    const typing = document.getElementById(id);
    if(typing) typing.remove();
}

// 打字效果
async function typeMessage(text) {
    const messageDiv = document.createElement('div');
    messageDiv.className = 'chat-message bot';
    chatContainer.appendChild(messageDiv);
    
    for(let i = 0; i < text.length; i++) {
        messageDiv.innerText += text[i];
        chatContainer.scrollTop = chatContainer.scrollHeight;
        await new Promise(resolve => setTimeout(resolve, 50));
    }
}

// 回车发送
chatInput.addEventListener('keydown', function(event) {
    if(event.key === 'Enter') {
        sendMessage();
    }
});

七、总结与最佳实践

7.1 事件处理最佳实践

  • 优先使用 addEventListener 绑定事件
  • 善用 事件委托 提高性能
  • 及时 移除事件监听 避免内存泄漏
  • 使用 event.target 获取触发元素

7.2 异步编程演进

  • 回调函数：基础但容易产生回调地狱
  • Promise：链式调用，更优雅
  • async/await：语法糖，代码最清晰

7.3 异步编程最佳实践

  • 现代开发优先使用 async/await
  • 始终处理错误：try-catch 或 .catch()
  • 可以并行的操作使用 Promise.all()
  • 避免在循环中使用 await (会串行执行)
