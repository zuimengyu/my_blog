8.2 HTTP 请求头/请求体/Cookie 协同处理

服务端接口需协同解析 HTTP 三大核心数据载体，完成业务逻辑：

// 完整请求数据处理示例（登录接口）
app.post('/login', (req, res) => {
  // 1. 解析请求体（用户名/密码）- 需 body-parser 中间件
  const { username, password } = req.body;
  
  // 2. 解析请求头（可选：客户端版本、设备信息等）
  const clientVersion = req.headers['x-client-version'];
  console.log('客户端版本：', clientVersion);
  
  // 3. 解析 Cookie（可选：记住登录状态）
  const rememberMe = req.cookies.rememberMe === 'true';
  
  // 4. 业务逻辑：验证用户名密码
  const isSuccess = username === 'admin' && password === '123456';
  
  // 5. 响应处理：设置 Cookie + 返回 Token
  if (isSuccess) {
    const token = jwt.sign({ username }, JWT_SECRET, { expiresIn: '24h' });
    // 设置 Cookie（记住登录状态）
    if (rememberMe) {
      res.cookie('rememberMe', 'true', { maxAge: 3600000 * 24 * 7 });
    }
    res.cookie('user', username, { maxAge: 3600000 * 24 });
    res.json({ success: true, token });
  } else {
    res.json({ success: false, message: '用户名或密码错误' });
  }
});

8.4 CORS 跨域问题完整解决方案

        
跨域是前后端分离架构的核心问题，server.js 通过自定义中间件彻底解决：

        


// 完整跨域中间件配置
app.use((req, res, next) => {
  // 1. 允许指定前端域名跨域（生产环境严格限制，避免任意域名访问）
  res.setHeader('Access-Control-Allow-Origin', FRONTEND_URL);
  
  // 2. 允许的请求方法（需包含 OPTIONS 预检请求）
  res.setHeader('Access-Control-Allow-Methods', 'GET, POST, OPTIONS');
  
  // 3. 允许的请求头（包含 Authorization 用于 Token 传递）
  res.setHeader('Access-Control-Allow-Headers', 'Content-Type, Authorization, X-Client-Version');
  
  // 4. 允许跨域请求携带 Cookie（双重验证必备）
  res.setHeader('Access-Control-Allow-Credentials', 'true');
  
  // 5. 预检请求（OPTIONS）直接响应 200，无需进入后续逻辑
  if (req.method === 'OPTIONS') {
    return res.status(200).end();
  }
  
  next(); // 非 OPTIONS 请求，进入接口逻辑
});

        

        

            💡 跨域配置关键原则：生产环境必须指定具体前端域名（如 https://xxx.com），禁止设置为 '*'（允许所有域名，存在安全风险）。

Node.js 核心优势是「非阻塞异步 I/O」，服务端所有 I/O 操作（文件、数据库、网络）均为异步，核心编程模型：


        
9.1 Promise 异步基础

        
Promise 是异步操作的标准化容器，解决回调嵌套问题，核心状态：

        
    
            
  • pending：初始状态，异步操作未完成
    • 
            
  • fulfilled：异步操作成功
    • 
            
  • rejected：异步操作失败
    • 
        

        


// Promise 封装异步操作（文件读取示例）
function readFileAsync(filePath) {
  return new Promise((resolve, reject) => {
    fs.readFile(filePath, 'utf8', (err, data) => {
      if (err) {
        reject(err); // 失败：调用 reject
      } else {
        resolve(data); // 成功：调用 resolve
      }
    });
  });
}

// 使用 Promise
readFileAsync('config.json')
  .then(data => {
    const config = JSON.parse(data);
    console.log('配置读取成功：', config);
  })
  .catch(err => {
    console.error('配置读取失败：', err);
  });

        


        
9.2 async/await 异步语法（服务端首选）

        
async/await 是 Promise 的语法糖，让异步代码写法接近同步，是 server.js 异步编程的核心：

        


// 1. 异步函数定义：async 关键字
app.post('/query', authMiddleware, async (req, res) => {
  try {
    // 2.  await 关键字：等待 Promise 完成（必须在 async 函数内）
    const [rows] = await db.execute(sql); // 等待数据库查询完成
    const excelPath = await generateExcel(rows); // 等待 Excel 生成完成
    res.json({ success: true, downloadUrl: excelPath });
  } catch (err) {
    // 3. 错误捕获：所有 await 异步操作的错误都能被 try/catch 捕获
    console.error('异步操作异常：', err);
    res.status(500).json({ success: false, message: '操作失败' });
  }
});

// 封装异步 Excel 生成函数
async function generateExcel(data) {
  const worksheet = xlsx.utils.json_to_sheet(data);
  const workbook = xlsx.utils.book_new();
  xlsx.utils.book_append_sheet(workbook, worksheet, '结果');
  
  const filename = `result-${Date.now()}.xlsx`;
  const filepath = path.join(__dirname, 'public', filename);
  
  // 等待文件写入完成（xlsx.writeFile 无 Promise 版本，需手动封装）
  await new Promise((resolve, reject) => {
    xlsx.writeFile(workbook, filepath, (err) => {
      err ? reject(err) : resolve(filepath);
    });
  });
  
  return `${FRONTEND_URL}/${filename}`;
}

        

        

            💡 服务端异步编程原则：所有 I/O 操作（数据库、文件、网络）必须用 async/await 处理，避免回调嵌套（回调地狱），同时通过 try/catch 统一捕获错误。
        


        
9.3 异步任务协同（定时器 + 异步 I/O）

        
server.js 中常需结合定时器与异步 I/O 完成复杂任务（如临时文件清理）：

        


// 异步清理过期临时文件（结合定时器与文件读取）
async function cleanExpiredFiles() {
  const publicDir = path.join(__dirname, 'public');
  
  try {
    // 异步读取 public 目录下的所有文件
    const files = await fs.promises.readdir(publicDir);
    
    // 遍历文件，筛选 Excel 临时文件
    for (const file of files) {
      if (file.startsWith('result-') && file.endsWith('.xlsx')) {
        const filePath = path.join(publicDir, file);
        // 异步获取文件创建时间
        const stats = await fs.promises.stat(filePath);
        const createTime = stats.birthtime;
        const now = new Date();
        const diffHours = (now - createTime) / (1000 * 60 * 60);
        
        // 超过 10 小时的文件异步删除
        if (diffHours > 10) {
          await fs.promises.unlink(filePath);
          console.log(`清理过期文件：${file}`);
        }
      }
    }
  } catch (err) {
    console.error('文件清理异常：', err);
  }
}

// 每小时执行一次清理任务
setInterval(cleanExpiredFiles, 3600000);

        


        
        
十、错误处理与日志管理

        
服务端错误处理直接影响稳定性，server.js 采用「分层捕获 + 日志持久化」方案，确保错误可追溯、服务不崩溃：


        
10.1 局部错误捕获（try/catch + 异步错误）

        
针对单个接口/函数的错误，通过 try/catch 精准捕获，避免错误扩散：

        


// 接口级错误捕获（async/await + try/catch）
app.post('/export', authMiddleware, async (req, res) => {
  const { data } = req.body;
  
  try {
    // 1. 参数校验（同步错误）
    if (!data || data.length === 0) {
      return res.status(400).json({ success: false, message: '无数据可导出' });
    }
    
    // 2. 异步 Excel 生成（异步错误）
    const worksheet = xlsx.utils.json_to_sheet(data);
    const workbook = xlsx.utils.book_new();
    xlsx.utils.book_append_sheet(workbook, worksheet, '查询结果');
    
    const filename = `result-${Date.now()}.xlsx`;
    const filepath = path.join(__dirname, 'public', filename);
    
    // 等待文件写入完成
    await new Promise((resolve, reject) => {
      xlsx.writeFile(workbook, filepath, (err) => {
        err ? reject(err) : resolve();
      });
    });
    
    res.json({ success: true, downloadUrl: `${FRONTEND_URL}/${filename}` });
  } catch (err) {
    // 3. 统一错误处理：记录日志 + 返回 500
    const errorMsg = `[导出接口异常] 用户：${req.user.username} | 错误：${err.message} | 堆栈：${err.stack}`;
    console.error(errorMsg);
    errorLogStream.write(`${new Date().toISOString()} | ${errorMsg}\n`);
    res.status(500).json({ success: false, message: '导出失败' });
  }
});

        


        
10.2 Express 全局错误中间件（兜底处理）

        
未被局部捕获的错误（如中间件错误、未处理的异步异常），通过全局中间件兜底，避免服务崩溃：

        


// 全局错误中间件（必须放在所有路由和中间件之后）
app.use((err, req, res, next) => {
  // 1. 格式化错误信息（包含关键上下文）
  const errorInfo = {
    time: new Date().toISOString(),
    ip: req.ip,
    path: req.path,
    method: req.method,
    body: req.body,
    message: err.message,
    stack: err.stack
  };
  
  // 2. 控制台输出 + 文件持久化
  console.error('[全局未捕获异常]', errorInfo);
  errorLogStream.write(`${JSON.stringify(errorInfo)}\n`);
  
  // 3. 统一响应格式（隐藏敏感错误信息）
  res.status(500).json({
    success: false,
    message: '服务器内部错误，请稍后重试'
  });
});

        

        

            ⚠️ 全局错误中间件必须放在所有路由定义之后，否则无法捕获路由和中间件中的错误！
        


        
10.3 分级日志管理（访问日志 + 错误日志）

        
日志是问题排查的核心依据，server.js 按级别分类管理，便于快速定位问题：

        


// 日志分级配置（server.js 核心日志逻辑）
const logDir = path.join(__dirname, 'logs');
if (!fs.existsSync(logDir)) fs.mkdirSync(logDir);

// 1. 错误日志：仅记录异常（500 错误、认证失败等）

// 2. 访问日志：仅记录状态码 ≥400 的异常请求（401 未授权、400 参数错误等）

app.use(morgan(':remote-addr [:date[ISO]] ":method :url" :status :response-time ms ":user-agent"', {

  skip: (req, res) => res.statusCode < 400 // 跳过正常请求
}));

// 3. 控制台日志：输出所有请求（开发环境便于调试）

  logStream.write(logStr);
  logStream.end(); // 关闭流，避免句柄泄漏
}

// 业务日志使用示例（登录成功记录）

11.2 Excel 导出完整流程（JSON →  文件 → 下载链接）

        
server.js 中文件导出的核心是「JSON 数据 → 本地文件 → 前端可访问 URL」的完整链路：

        


// Excel 导出完整实现（server.js 核心逻辑）
app.post('/export', authMiddleware, async (req, res) => {
  const { data } = req.body;
  
  try {
    // 1. 数据校验
    if (!data || !Array.isArray(data) || data.length === 0) {
      return res.status(400).json({ success: false, message: '无有效数据' });
    }
    
    // 2. JSON 转 Excel 工作表
    const worksheet = xlsx.utils.json_to_sheet(data);
    
    // 3. 创建 Excel 工作簿（相当于一个 .xlsx 文件）
    const workbook = xlsx.utils.book_new();
    xlsx.utils.book_append_sheet(workbook, worksheet, '查询结果'); // 工作表命名
    
    // 4. 生成唯一文件名（避免冲突）
    const filename = `result-${Date.now()}-${req.user.username}.xlsx`;
    // 5. 拼接文件存储路径（放入 public 静态目录）
    const filepath = path.join(__dirname, 'public', filename);
    
    // 6. 写入文件到本地
    await new Promise((resolve, reject) => {
      xlsx.writeFile(workbook, filepath, (err) => {
        if (err) reject(err);
        else resolve();
      });
    });
    
    // 7. 生成前端可访问的下载链接（静态资源 URL）
    const downloadUrl = `${FRONTEND_URL}/${filename}`;
    
    // 8. 10 分钟后自动清理临时文件
    setTimeout(() => {
      if (fs.existsSync(filepath)) {
        fs.unlinkSync(filepath);
        console.log(`清理临时 Excel 文件：${filename}`);
      }
    }, 600000);
    
    // 9. 返回下载链接给前端
    res.json({
      success: true,
      downloadUrl: downloadUrl,
      message: `Excel 导出成功，链接有效期 10 分钟`
    });
  } catch (err) {
    console.error('Excel 导出异常：', err);
    res.status(500).json({ success: false, message: '导出失败' });
  }
});

十二、接口开发规范

        
规范的接口设计是服务端可维护性的核心，server.js 遵循行业通用规范，确保接口易用、易调试：


        
12.1 接口参数接收与严格校验

        
所有接口必须对参数进行校验，避免无效参数导致的业务异常：

        


// 接口参数校验示例（/query 接口）
app.post('/query', authMiddleware, async (req, res) => {
  const { type, duration, status } = req.body;
  
  // 1. 必传参数校验
  const requiredParams = ['type', 'duration', 'status'];
  const missingParams = requiredParams.filter(key => !req.body[key]);
  if (missingParams.length > 0) {
    return res.status(400).json({
      success: false,
      message: `缺少必填参数：${missingParams.join(',')}`
    });
  }
  
  // 2. 参数格式校验
  const validTypes = ['私教', '普通'];
  const validDurations = ['15', '30'];
  const validStatuses = ['沉睡', '到期'];
  
  if (!validTypes.includes(type)) {
    return res.status(400).json({ success: false, message: 'type 只能是 私教/普通' });
  }
  if (!validDurations.includes(duration.toString())) {
    return res.status(400).json({ success: false, message: 'duration 只能是 15/30' });
  }
  if (!validStatuses.includes(status)) {
    return res.status(400).json({ success: false, message: 'status 只能是 沉睡/到期' });
  }
  
  // 3. 校验通过，执行查询逻辑
  let sql = '';
  switch (`${type}-${duration}-${status}`) {
    case '私教-30-沉睡':
      sql = 'SELECT DISTINCT tb.store,tb.sales FROM 私教合同 tb LEFT JOIN 上课日志 ta ON tb.Member_Phone=ta.Member_Phone WHERE ta.Member_Phone IS NULL';
      break;
    // 其他 case...
  }
  
  const [rows] = await db.execute(sql);
  res.json({ success: true, data: rows, total: rows.length });
});

14.1 临时文件自动清理

        
导出的 Excel 等临时文件需定时清理，避免磁盘空间被占满：

        


// 临时文件清理方案（server.js 核心实现）
const EXCEL_DIR = path.join(__dirname, 'public');
const EXPIRE_HOURS = 1; // 临时文件有效期 1 小时

// 定时清理函数
function cleanTempFiles() {
  try {
    // 读取 public 目录下的 Excel 文件
    const files = fs.readdirSync(EXCEL_DIR);
    const now = new Date();
    
    files.forEach(file => {
      if (file.startsWith('result-') && file.endsWith('.xlsx')) {
        const filePath = path.join(EXCEL_DIR, file);
        const stats = fs.statSync(filePath);
        const fileAgeHours = (now - stats.birthtime) / (1000 * 60 * 60);
        
        // 删除过期文件
        if (fileAgeHours > EXPIRE_HOURS) {
          fs.unlinkSync(filePath);
          console.log(`清理过期临时文件：${file}（已存在 ${fileAgeHours.toFixed(1)} 小时）`);
        }
      }
    });
  } catch (err) {
    console.error('临时文件清理异常：', err);
  }
}

// 启动时执行一次清理
cleanTempFiles();

// 每 30 分钟执行一次清理
setInterval(cleanTempFiles, 30 * 60 * 1000);

        


        
14.2 数据库连接池优化

// 数据库连接池优化配置
const pool = mysql.createPool({
  host: DB_HOST,
  user: DB_USER,
  password: DB_PASSWORD,
  database: DB_DATABASE,
  port: 3306,
  connectionLimit: 20, // 最大连接数（根据数据库性能调整，默认 10）
  waitForConnections: true, // 无空闲连接时等待（避免直接报错）

  enableKeepAlive: true, // 启用长连接
  keepAliveInitialDelay: 300000 // 5 分钟发送一次心跳包，保持连接活跃
});

// 连接池状态监控（可选）
setInterval(async () => {
  const [rows] = await pool.execute('SELECT 1');
  const connections = pool._allConnections.length;
  const freeConnections = pool._freeConnections.length;

14.3 流式操作与内存优化

        
大文件（日志、Excel）操作采用流式处理，避免一次性加载到内存导致内存溢出：

        


// 流式日志写入（避免一次性写入大日志占用内存）
const accessLogStream = fs.createWriteStream(path.join(LOG_DIR, 'access.log'), { flags: 'a' });
// morgan 日志直接写入流，而非先缓存
app.use(morgan(':remote-addr [:date[ISO]] ":method :url" :status :response-time ms', {
  stream: accessLogStream
}));

// 大 Excel 文件流式生成（xlsx 模块优化）
function generateLargeExcel(data, filename) {
  return new Promise((resolve, reject) => {
    const filepath = path.join(PUBLIC_DIR, filename);
    const workbook = xlsx.utils.book_new();
    const worksheet = xlsx.utils.json_to_sheet(data);
    xlsx.utils.book_append_sheet(workbook, worksheet, '大数据量结果');
    
    // 流式写入文件（xlsx.writeFile 内部优化）
    xlsx.writeFile(workbook, filepath, (err) => {
      if (err) reject(err);
      else resolve(filepath);
    });
  });
}
